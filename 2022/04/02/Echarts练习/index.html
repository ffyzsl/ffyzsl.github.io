建立数据库

打开 MySQL，或者使用 Navicat for MySQL

  • 建立 school_information 数据库
    • MySQL 语句：create database school_information
  • 建立 相关表 MySQL 语句：CREATE TABLE <表名> ([表定义选项])[表选项][分区选项];
    • grades ：成绩表
    • school：学校表
    • student：学生表
    • student_info：学生信息表
    • student_intern：学生实习表
    • teacher：教师表
    • upmassage：用户信息表

新建项目

第一步 ：打开 IDEA 点击 NewProject

第二步：点击 Maven ，勾选 Create from archetype，选择 org.apache.maven.archetypes:maven=archetype-webapp，点击 Next

第三步：在 Name 处为项目命名，并选择 存储路径，完成后点击 Next

第四步：

  • Maven home path 选择下载maven包的路径
  • User settings file 选择maven 配置文件
  • local repository 选择本地仓库

完成后点击 Next

配置相关依赖

在项目目录处选择 pom.xml，以下代码在 dependencies标签中完成

配置 Tomcat

第一步：点击界面上方的 Add Configuration…

第二步：点击弹出的界面的左上方的加号

第三步：点击 Tomcat Server下的 local（Community版本没有此功能）

第四步：点击 Configure…,找到本地目录下 tomcate 文件夹。（如果界面下方出现 Warning，点击 Fix 修复即可），完成后点击 OK 或 Apply

创建目录及相关包

因为使用了 Maven模板 ，所以自动生成以下目录。

右键 main ，选择 New 里的 Directory，创建名称为 java 的文件夹

右键java文件夹，选择 Mark Directory as 里的 Sources Root

依次创建 action ,dao,entity,servlet 包,在dao包下创建 student_infor包

  • action：把数据转化为 JSON 格式
  • dao：功能实现类
    • student_infor：学生信息实现类
  • entity：实体类
  • servlet：所有servlet类

在 webapp下新建 css,pic 文件夹

  • css：存放css文件
  • pic：存放照片

      package action;

      import entity.School;
      import entity.Student;
      import entity.Teacher;
      import net.sf.json.JSONArray;
      import net.sf.json.JSONObject;

      import java.util.ArrayList;

      public class Controller {

      public static void getRelation_schoolTeacher(Teacher o1,School o2,JSONArray jsonArray){
    JSONObject jsonObject=new JSONObject();
    jsonObject.put("source",o1.getTeacher_name());
    jsonObject.put("target",o2.getSchool_name());
    jsonArray.add(jsonObject);
}

public static void getRelation_teacherStudent(Student o1,Teacher o2,JSONArray jsonArray){
    JSONObject jsonObject=new JSONObject();
    jsonObject.put("source",o1.getStudent_name());
    jsonObject.put("target",o2.getTeacher_name());
    jsonArray.add(jsonObject);
}


public static JSONArray getObject(ArrayList<Object> arrayList){
    JSONArray jsonArray=new JSONArray();

    for (int i=0;i<arrayList.size();i++){
        if (arrayList.get(i) instanceof School){
            School school=(School) arrayList.get(i);
            JSONObject jsonObject=new JSONObject();
            jsonObject.put("name",school.getSchool_name());
            jsonArray.add(jsonObject);
        }
        if (arrayList.get(i) instanceof Teacher){
            Teacher teacher=(Teacher) arrayList.get(i);
            JSONObject jsonObject=new JSONObject();
            jsonObject.put("name",teacher.getTeacher_name());
            jsonArray.add(jsonObject);
        }
        if (arrayList.get(i) instanceof Student){
            Student student=(Student) arrayList.get(i);
            JSONObject jsonObject=new JSONObject();

            jsonObject.put("name",student.getStudent_name());
            jsonObject.put("des",student.getStudent_info());

            /* 判断学生成绩是否及格
             * pass: 及格
             * fail: 不及格
             */
            if (student.getFlag()==false) {
                jsonObject.put("category","fail");
            }else {
                jsonObject.put("category","pass");
            }

            jsonArray.add(jsonObject);
        }
    }
    return jsonArray;
}

      }

    • public void getObject(String id,ArrayList<Object> arrayList,JSONArray jsonArray) throws SQLException：入口方法，通过该方法调用 学校查找方法，教师查找方法，学生查找方法。

      package dao;import action.Controller;
      import entity.School;
      import net.sf.json.JSONArray;import java.sql.Connection;
      import java.sql.ResultSet;
      import java.sql.SQLException;
      import java.sql.Statement;
      import java.util.ArrayList;public class QuerySchoolDao {
      public static void getSchoolResult(String id, ArrayList<Object> arrayList, Connection connection, JSONArray jsonArray){

    ResultSet resultSet_school=null;

    String sql="select * from school where school_id = '"+id+"';";

    try {
        Statement statement=connection.createStatement();
        resultSet_school=statement.executeQuery(sql);

        while(resultSet_school.next()){
            School school=new School();
            school.setSchool_id(resultSet_school.getString("school_id"));
            school.setSchool_name(resultSet_school.getString("school_name"));
            arrayList.add(school);//把查询到的数据存放在 arrayList 中

            QueryTeacherDao.getTeacher(school,arrayList,connection,jsonArray);//调用查询教师方法

            System.out.println("school getTeacher 之后");
        }


    } catch (SQLException e) {
        e.printStackTrace();
    }


}
      }

        while(resultSet.next()){
            Student student=new Student();
            student.setTeacher_name(resultSet.getString("teacher_name"));
            student.setStudent_name(resultSet.getString("student_name"));

        jsonArrayLink.add(jsonObject);
